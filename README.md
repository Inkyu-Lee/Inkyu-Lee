public class WebConfig implements WebMvcConfigurer {

     * @param registry
     */
    @Override

        registry.addMapping("/**")
                .allowedOriginPatterns("*")
                .allowedMethods("*")
                .allowedHeaders("*")
                .allowCredentials(true)
                .maxAge(3600);
